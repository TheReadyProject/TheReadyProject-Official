Atividade:

Criação de um programa que gerencia uma lista de compras, permitindo adicionar, remover e visualizar itens.

#include <vector>
#include <iostream>
#include <string>

int main() {
    std::vector<std::string> listaCompras;
    std::string item;
    int opcao;
    
    do {
        std::cout << "\nLista de Compras:" << std::endl;
        for (const auto& i : listaCompras) {
            std::cout << "- " << i << std::endl;
        }
        
        std::cout << "\nOpções:" << std::endl;
        std::cout << "1. Adicionar item" << std::endl;
        std::cout << "2. Remover item" << std::endl;
        std::cout << "3. Sair" << std::endl;
        std::cout << "Escolha uma opção: ";
        std::cin >> opcao;
        
        std::cin.ignore();  // Limpar buffer do \n
        
        if (opcao == 1) {
            std::cout << "Digite o item: ";
            std::getline(std::cin, item);
            listaCompras.push_back(item);
        } else if (opcao == 2) {
            std::cout << "Digite o item a remover: ";
            std::getline(std::cin, item);
            auto it = std::find(listaCompras.begin(), listaCompras.end(), item);
            if (it != listaCompras.end()) {
                listaCompras.erase(it);
            } else {
                std::cout << "Item não encontrado." << std::endl;
            }
        } else if (opcao != 3) {
            std::cout << "Opção inválida." << std::endl;
        }
    } while (opcao != 3);

    return 0;
}

Atividade:

Criação de uma calculadora simples com funções para adição, subtração, multiplicação e divisão.

#include <iostream>

int adicionar(int a, int b) {
    return a + b;
}

int subtrair(int a, int b) {
    return a - b;
}

int multiplicar(int a, int b) {
    return a * b;
}

double dividir(int a, int b) {
    if (b == 0) {
        throw std::invalid_argument("Divisão por zero não permitida");
    }
    return static_cast<double>(a) / b;
}

int main() {
    int num1, num2, escolha;
    
    std::cout << "Digite o primeiro número: ";
    std::cin >> num1;
    std::cout << "Digite o segundo número: ";
    std::cin >> num2;
    
    std::cout << "Escolha a operação:" << std::endl;
    std::cout << "1. Adição" << std::endl;
    std::cout << "2. Subtração" << std::endl;
    std::cout << "3. Multiplicação" << std::endl;
    std::cout << "4. Divisão" << std::endl;
    std::cin >> escolha;
    
    try {
        switch (escolha) {
            case 1:
                std::cout << "Resultado: " << adicionar(num1, num2) << std::endl;
                break;
            case 2:
                std::cout << "Resultado: " << subtrair(num1, num2) << std::endl;
                break;
            case 3:
                std::cout << "Resultado: " << multiplicar(num1, num2) << std::endl;
                break;
            case 4:
                std::cout << "Resultado: " << dividir(num1, num2) << std::endl;
                break;
            default:
                std::cout << "Operação inválida." << std::endl;
        }
    } catch (const std::invalid_argument& e) {
        std::cerr << e.what() << std::endl;
    }
    
    return 0;
}

Encapsulamento:

#include <iostream>

class Pessoa {
private:
    std::string nome;
    int idade;

public:
    Pessoa(std::string n, int i) : nome(n), idade(i) {}

    std::string getNome() {
        return nome;
    }

    void setNome(std::string n) {
        nome = n;
    }

    int getIdade() {
        return idade;
    }

    void setIdade(int i) {
        idade = i;
    }

    void apresentar() {
        std::cout << "Olá, meu nome é " << nome << " e tenho " << idade << " anos." << std::endl;
    }
};

int main() {
    Pessoa pessoa("Alice", 30);
    pessoa.apresentar();
    pessoa.setNome("Bob");
    pessoa.setIdade(25);
    pessoa.apresentar();
    return 0;
}

Atividade:

Criação de uma classe `Carro` com atributos (marca, modelo, ano) e métodos (exibir informações, acelerar, frear).

#include <iostream>

class Carro {
private:
    std::string marca;
    std::string modelo;
    int ano;

public:
    Carro(std::string m, std::string mo, int a) : marca(m), modelo(mo), ano(a) {}

    void exibirInformacoes() {
        std::cout << "Marca: " << marca << ", Modelo: " << modelo << ", Ano: " << ano << std::endl;
    }

    void acelerar() {
        std::cout << "O carro está acelerando." << std::endl;
    }

    void frear() {
        std::cout << "O carro está freando." << std::endl;
    }
};

int main() {
    Carro carro("Toyota", "Corolla", 2020);
    carro.exibirInformacoes();
    carro.acelerar();
    carro.frear();
    return 0;
}
